607  template<typename ScoresItT> static double getROCAuc(
608  ScoresItT posScoresBegin, ScoresItT posScoresEnd,
609  ScoresItT negScoresBegin, ScoresItT negScoresEnd, bool parallel =
610  false) {
611 
612  std::size_t nbLess = 0;
613  std::size_t nbEq = 0;
614 
615  std::size_t P = posScoresEnd - posScoresBegin;
616  std::size_t N = negScoresEnd - negScoresBegin;
617 
618  // The smallest range is sorted
619  if (P <= N) {
620 
621 #ifdef LINKPRED_WITH_OPENMP
622 #pragma omp parallel for reduction(+ : nbLess), reduction(+ : nbEq) if (parallel)
623 #endif
624  for (auto pit = negScoresBegin; pit < negScoresEnd; ++pit) {
625  auto itl = std::lower_bound(posScoresBegin, posScoresEnd, *pit);
626  auto itu = std::upper_bound(posScoresBegin, posScoresEnd, *pit);
627  nbLess += posScoresEnd - itu;
628  nbEq += itu - itl;
629  }
630  } else {
631 
632 #ifdef LINKPRED_WITH_OPENMP
633 #pragma omp parallel for reduction(+ : nbLess), reduction(+ : nbEq) if (parallel)
634 #endif
635  for (auto pit = posScoresBegin; pit < posScoresEnd; ++pit) {
636  auto itl = std::lower_bound(negScoresBegin, negScoresEnd, *pit);
637  auto itu = std::upper_bound(negScoresBegin, negScoresEnd, *pit);
638  nbLess += itl - negScoresBegin;
639  nbEq += itu - itl;
640  }
641 
642  }
643 
644  return (nbLess + 0.5 * nbEq) / (P * N);
645  }

711  template<typename StrmScoresItT, typename StoredScoresItT> double getROCAucStrm(
712  StrmScoresItT strmBegin, StrmScoresItT strmEnd,
713  StoredScoresItT strdBegin, StoredScoresItT strdEnd) {
714 
715  std::size_t nbStrm = 0;
716  auto beginSP = std::make_shared<StrmScoresItT>(strmBegin);
717  auto endSP = std::make_shared<StrmScoresItT>(strmEnd);
718 #ifdef LINKPRED_WITH_MPI
719  int nbProcs = 1;
720  int procID = 0;
721  if (distributed) {
722  MPI_Comm_size(comm, &nbProcs);
723  MPI_Comm_rank(comm, &procID);
724  nbStrm = strmEnd - strmBegin;
725  auto lr = Utils::localRange(nbStrm, nbProcs, procID);
726  beginSP = std::make_shared<StrmScoresItT>(strmBegin + lr.first);
727  endSP = std::make_shared<StrmScoresItT>(strmBegin + lr.second);
728  }
729 #endif
730 
731  auto begin = *beginSP;
732  auto end = *endSP;
733 
734  std::size_t nbLess = 0;
735  std::size_t nbEq = 0;
736 
737  std::size_t nbStrd = strdEnd - strdBegin;
738 
739  if (nbStrm == 0) {
740 #ifdef LINKPRED_WITH_OPENMP
741 #pragma omp parallel for reduction(+ : nbStrm), reduction(+ : nbLess), reduction(+ : nbEq) if (parallel)
742  for (auto pit = begin; pit < end; ++pit) {
743 #else
744  for (auto pit = begin; pit != end; ++pit) {
745 #endif
746  double sc = pit->second;
747  auto itl = std::lower_bound(strdBegin, strdEnd, sc);
748  auto itu = std::upper_bound(strdBegin, strdEnd, sc);
749  nbStrm++;
750  nbLess += strdEnd - itu;
751  nbEq += itu - itl;
752  }
753  } else {
754 #ifdef LINKPRED_WITH_OPENMP
755 #pragma omp parallel for reduction(+ : nbLess), reduction(+ : nbEq) if (parallel)
756  for (auto pit = begin; pit < end; ++pit) {
757 #else
758  for (auto pit = begin; pit != end; ++pit) {
759 #endif
760  double sc = pit->second;
761  auto itl = std::lower_bound(strdBegin, strdEnd, sc);
762  auto itu = std::upper_bound(strdBegin, strdEnd, sc);
763  nbLess += strdEnd - itu;
764  nbEq += itu - itl;
765  }
766  }
767 
768 #ifdef LINKPRED_WITH_MPI
769  if (distributed) {
770  unsigned long long lNbLessEq[2];
771  unsigned long long gNbLessEq[2];
772  lNbLessEq[0] = nbLess;
773  lNbLessEq[1] = nbEq;
774  MPI_Reduce(lNbLessEq, gNbLessEq, 2, MPI_UNSIGNED_LONG_LONG, MPI_SUM,
775  0, comm);
776  nbLess = gNbLessEq[0];
777  nbEq = gNbLessEq[1];
778  }
779 #endif
780  return (nbLess + 0.5 * nbEq) / (nbStrd * nbStrm);
781  }
