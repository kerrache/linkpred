112  virtual double score(Edge const & e) {
113  auto srcNode = Network::start(e);
114  auto endNode = Network::end(e);
115 
116  std::vector<typename Network::Edge> ce; // Common neighbors of both nodes
117  net->getCommonNeighbors(srcNode, endNode, std::back_inserter(ce));
118  std::vector<typename Network::NodeID> ns;
119  for (auto cnit = ce.cbegin(); cnit != ce.cend(); ++cnit) {
120  ns.push_back(Network::end(*cnit));
121  }
122  std::vector<std::size_t> degs; // Internal degree of each of the neighbors
123  for (auto it1 = ns.cbegin(); it1 != ns.cend(); ++it1) {
124  std::size_t nb = 0;
125  for (auto it2 = ns.cbegin(); it2 != ns.cend(); ++it2) {
126  if (net->isEdge(*it1, *it2)) {
127  nb++;
128  }
129  }
130  degs.push_back(nb);
131  }
132  double sc = 0;
133  for (std::size_t i = 0; i < ns.size(); i++) {
134  sc += (double) degs[i] / net->getDeg(ns[i]);
135  }
136  return sc;
137  }
