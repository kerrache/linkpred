362  virtual double score(Edge const & e) {
363  auto coordsi = nodesCoord.at(Network::start(e));
364  auto coordsj = nodesCoord.at(Network::end(e));
365 
366  double dtheta = std::abs(coordsi.first - coordsj.first);
367  if (dtheta > MathPI) {
368  dtheta = 2 * MathPI - dtheta;
369  }
370  double ri = coordsi.second;
371  double rj = coordsj.second;
372  double dist = (1 / zeta)
373  * std::acosh(
374  (std::cosh(zeta * ri) * std::cosh(zeta * rj))
375  - (std::sinh(zeta * ri) * std::sinh(zeta * rj)
376  * std::cos(dtheta)));
377  return 1 / (1 + std::exp((zeta / (2 * T)) * dist));
378  }
