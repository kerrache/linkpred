85  std::size_t topDegMonotone(std::size_t l, EdgeRndOutIt eit,
86  ScoreRndIt sit) {

88  // Copy and sort degrees
89  std::vector<std::pair<NodeID, std::size_t>> deg;
90  deg.resize(net->getNbNodes());
91  for (auto it = net->nodesDegBegin(); it < net->nodesDegEnd(); ++it) {
92  deg[it - net->nodesDegBegin()] = *it;
93  }
94  std::sort(deg.begin(), deg.end(),
95  [](auto a, auto b) {return a.second > b.second;});

100  struct Comparator {
101  bool operator ()(
102  std::pair<double, std::pair<std::size_t, std::size_t>> const & left,
103  std::pair<double, std::pair<std::size_t, std::size_t>> const & right) {
104  return left.first < right.first;
105  }
106  };
107 
108  std::priority_queue<
109  std::pair<double, std::pair<std::size_t, std::size_t>>,
110  std::vector<
111  std::pair<double, std::pair<std::size_t, std::size_t>>>,
112  Comparator> pq;
113  pq.push(
114  std::make_pair(
115  this->score(net->makeEdge(deg[0].first, deg[1].first)),
116  std::make_pair(0, 1)));
117  inq.insert(std::make_pair(0, 1));
118  std::size_t cpt = 0;
119  while (cpt < l && !pq.empty()) {
120  auto el = pq.top();
121  pq.pop();
122 // inq.erase(el.second); // Cannot erase here, otherwise we'll have repeated visits
123  auto ii = deg[el.second.first].first;
124  auto jj = deg[el.second.second].first;
125  if (!net->isEdge(ii, jj)) {
126  *eit = net->makeEdge(ii, jj);
127  ++eit;
128  *sit = el.first;
129  ++sit;
130  cpt++;
131  if (cpt == l) {
132  break;
133  }
134  }
135  std::size_t i = el.second.first;
136  std::size_t j = el.second.second;
137  if (j < nbNodes - 1) {
138  auto e = std::make_pair(i, j + 1);
139  if (inq.count(e) == 0) {
140  pq.push(
141  std::make_pair(
142  this->score(
143  net->makeEdge(deg[i].first,
144  deg[j + 1].first)), e));
145  inq.insert(e);
146  }
147  }
148  if (i < nbNodes - 1 && i + 1 < j) {
149  auto e = std::make_pair(i + 1, j);
150  if (inq.count(e) == 0) {
151  pq.push(
152  std::make_pair(
153  this->score(
154  net->makeEdge(deg[i + 1].first,
155  deg[j].first)),
156  std::make_pair(i + 1, j)));
157  inq.insert(e);
158  }
159  }
160  }
161 
162  return cpt;
163  }

346  }
347 }

350  // Computing loop start, end and step in case of distributed processing
351 #ifdef LINKPRED_WITH_MPI
352  int nbProcs;
353  int procID;
354  if (distributed) {
355  MPI_Comm_size(comm, &nbProcs);
356  MPI_Comm_rank(comm, &procID);
357  } else {
358  nbProcs = 1;
359  procID = 0;
360  }
361  auto start = net->nonEdgesBegin() + procID;
362  auto end = net->nonEdgesEnd();
363  auto step = nbProcs;
364 #else

366  auto end = net->nonEdgesEnd();
367  auto step = 1;
368 #endif
369 
370 #ifdef LINKPRED_WITH_OPENMP
371 #pragma omp parallel for if (parallel) schedule(dynamic) // We need a dynamic schedule because iteration vary greatly in duration
372 #endif
373  for (auto eit = start; eit < end; eit += step) {
374 #ifdef LINKPRED_WITH_OPENMP
375  int ind = omp_get_thread_num();
376 #else
377  int ind = 0;
378 #endif
379  mqs[ind].push(*eit, this->score(*eit));
380  }
381 
382 #ifdef LINKPRED_WITH_OPENMP
383  // Now we merge results obtained by all threads
384  if (parallel) {
385  LMapQueue<Edge, double>::parMerge(mqs.begin(), mqs.end());
386  }
387 #endif
388 
389 #ifdef LINKPRED_WITH_MPI
390  if (distributed) {
391  // Now we merge over all processes
392  Utils::merge(mqs[0], comm);
393  }
394 #endif
395 
396  for (auto it = mqs[0].begin(); it != mqs[0].end(); ++it) {
397  *eit = it->first;
398  ++eit;
399  *sit = it->second;
