135  inline double RA1EdgeLength(Edge const & edge) const {
136  auto srcNode = Network::start(edge);
137  auto endNode = Network::end(edge);
138  double ki = net->getDeg(srcNode);
139  double kj = net->getDeg(endNode);
140  double cn = net->getNbCommonNeighbors(srcNode, endNode);
141  return (ki + kj + ki * kj) / (1 + cn)

150  inline double RA2EdgeLength(Edge const & edge) const {
151  auto srcNode = Network::start(edge);
152  auto endNode = Network::end(edge);
153  double cn = net->getNbCommonNeighbors(srcNode, endNode);
154  double eij;
155  if (net->isEdge(srcNode, endNode)) {
156  eij = 1;
157  } else {
158  eij = 0;
159  }
160  double ei = net->getDeg(srcNode) - cn - eij;
161  double ej = net->getDeg(endNode) - cn - eij;
162  return (1 + ei + ej + ei * ej) / (1 + cn)
163  + std::numeric_limits<double>::epsilon();
164  }

262  virtual double score(Edge const & e) {
263 
264  double sc = 0;
265  auto srcNode = Network::start(e);
266  auto endNode = Network::end(e);
267  switch (edgeLengthMethod) {
268  case PAT: {
269  if (net->isEdge(srcNode, endNode)) {
270  auto res = distCalc->getIndDist(srcNode, endNode);
271  sc =
272  1.0
273  / (1
274  + std::pow(res.first, lambda)
275  * std::pow(PATEdgeLength(e),
276  lambda - 1));
277  } else {
278  auto res = distCalc->getDist(srcNode, endNode);
279  sc =
280  1.0
281  / (1
282  + std::pow(res.first, lambda)
283  * std::pow(PATEdgeLength(e),
284  lambda - 1));
285  }
286  break;
