207 template<typename RandomIterator> std::set<
208  typename std::iterator_traits<RandomIterator>::value_type> selectRandom(
209  RandomIterator begin, RandomIterator end, std::size_t k,
210  long int seed) {
211  logger(logDebug, "Selecting elements randomly...")
212  typedef typename std::iterator_traits<RandomIterator>::value_type T;
213  std::size_t n = std::distance(begin, end);
214  if (k > n) {
215  throw std::out_of_range(
216  "The number of elements to select is larger than the total number of elements.");
217  }
218  RandomGen rng(seed);
219  std::set<T> selected;
220  for (std::size_t i = n - k; i < n; i++) {
221  std::size_t ind = rng.getSInt(0, i - 1);
222  RandomIterator it = begin + ind;
223  if (selected.count(*it) == 0) {
224  selected.insert(*it);
225  } else {
226  selected.insert(*(begin + i));
227  }
228  }
229  logger(logDebug, "Done")
230  return selected;
231 }

285 template<typename T, typename InputIterator, typename InserterIt,
286  typename Compare> void selectTopK(InputIterator begin,
287  InputIterator end, InserterIt inserter, std::size_t k) {
288  std::size_t n = std::distance(begin, end);
289  if (k > n) {
290  throw std::out_of_range(
291  "The number of elements to select is larger than the total number of elements.");
292  }
293  std::priority_queue<T, std::vector<T>, Compare> pq;
294  auto it = begin;
295  for (std::size_t i = 0; i < k; i++) {
296  T val = *it;
297  pq.push(val);
298  ++it;
299  }
300  Compare compare;
301  for (; it != end; ++it) {
302  if (!compare(pq.top(), *it)) {
303  pq.pop();
304  pq.push(*it);
305  }
306  }
307  while (!pq.empty()) {
308  inserter = pq.top();
309  ++inserter;
310  pq.pop();
311  }
312 }

443 template<typename KComparator, typename PComparator> void merge(
444  LMapQueue<unsigned long long, double, KComparator, PComparator> & mq,
445  MPI_Comm const & comm) {
446  enum Tags {
447  MergeTag = 125478
448  };
449 
450  int nbProcs;
451  int procID;
452  MPI_Comm_size(comm, &nbProcs);
453  if (nbProcs == 1) {
454  return; // Nothing to do here
455  }
456  MPI_Comm_rank(comm, &procID);
457 
458 // Create an MPI structure for transferring queue content
459  int count = 2;
460  int blockLengths[2] = {1, 1};
461  MPI_Datatype types[2] = {MPI_UNSIGNED_LONG_LONG, MPI_DOUBLE};
462  MPI_Datatype mpiPairULLD;
463  typedef std::pair<unsigned long long, double> pairULLD;
464  MPI_Aint offsets[] =
465  { offsetof(pairULLD, first), offsetof(pairULLD, second)};
466  MPI_Type_create_struct(count, blockLengths, offsets, types, &mpiPairULLD);
467  MPI_Type_commit(&mpiPairULLD);
468 
469  for (int j = 0; j < std::ceil(std::log2(nbProcs)); j++) {
470  int d = std::pow(2, j);
471  int nd = std::pow(2, j + 1);
472  if (procID % nd == 0) { // Receiver
473  if (procID + d < nbProcs) {
474  std::vector<std::pair<std::size_t, double>> rmq;
475  rmq.resize(mq.getL()); // Space for receiving data
476  MPI_Status status;
477  MPI_Recv(rmq.data(), rmq.size(), mpiPairULLD, procID + d,
478  MergeTag, comm, &status);
479  int nbRec = 0;
480  MPI_Get_count(&status, mpiPairULLD, &nbRec);
481  // Merge received data with local queue
482  for (int i = nbRec - 1; i >= 0; i--) {
483  if (mq.size() == mq.getL()
484  && !mq.compareTop(rmq[i].second)) {
485  break;
486  }
487  mq.push(rmq[i]);
488  }
489  }
490  } else if (procID % d == 0) { // Sender
491  // Senders must sort their data
492  // Notice that the local queue will be destroyed, but
493  // this is OK, senders will not use their local data any more
494  std::vector<std::pair<std::size_t, double>> smq;
495  if (procID != 0) {
496  smq.reserve(mq.size());
497  while (!mq.empty()) {
498  smq.push_back(mq.top());
499  mq.pop();
500  }
501  }
502 // for(auto it = smq.begin(); it != smq.end(); ++it) {
503 // std::cout << "Merge: " << it->second << std::endl;
504 // }
505 
506  MPI_Send(smq.data(), smq.size(), mpiPairULLD, procID - d, MergeTag,
507  comm);
508  }
509  // Wait for all processes to finish
510  MPI_Barrier(comm);
511  }
512 }

522  std::string fileName, bool ignoreLoops = true) {
523 
524  logger(logDebug, "Reading edges from file...")

528  if (!in) {

531
